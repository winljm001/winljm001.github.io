# bigjs 加减乘除

import Big from "big.js";

// 加
0.1 + 0.2; // 0.30000000000000004
x = new Big(0.1);
y = x.plus(0.2);

// 减
0.3 - 0.1; // 0.19999999999999998
x = new Big(0.3);
x.minus(0.1);

// 乘
0.6 * 3;
x = new Big(0.6);
y = x.times(3);

// 除
x = new Big(355);
y = new Big(113);
x.div(y);

// 求幂
Math.pow(0.7, 2); // 0.48999999999999994
x = new Big(0.7);
x.pow(2); // '0.49'
