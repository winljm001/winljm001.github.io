# lodash 求并，交，差

import _ from "lodash";

// 并
_.union([2], [1, 2]);
// => [2, 1]

// 交
_.intersection([2, 1], [4, 2], [1, 2]);
// => [2]

// 差
_.difference([3, 2, 1], [4, 2]);
// => [3, 1]
